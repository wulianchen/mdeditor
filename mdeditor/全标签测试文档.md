```python
# 计算exe文件的MD5值
import hashlib

def calculate_md5(file_path):
    md5 = hashlib.md5()
    with open(file_path, 'rb') as f:
        while chunk := f.read(4096):
            md5.update(chunk)
    return md5.hexdigest()

# 调用函数

print(f"文件MD5值：{file_md5}")
```
